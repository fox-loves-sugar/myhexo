4 Vue端

4.1 UserOnline.vue

<template>
    
  <div>
        <a-button type="primary" @click="showModal">
            实时聊天     
        </a-button>
        <a-modal v-model="visible" title="聊天框" @ok="handleOk">                
          <div style="height: 100px">
            <p>历史记录：</p>
            <table>
              <tr v-for="(h, item) in history" :key="item">
                <td>{{h.username}}:</td>
                <td>{{h.message}}</td>
              </tr>
            </table>
          </div>           
          <p><a-textarea style="height: 100px" v-model="message"></a-textarea></p>           
          <p><a-button style="float: right" @click="sendMessage">发送消息</a-button></p>
          <br />
        </a-modal>     
  </div>
</template>
<script>
export default {
  data() {
    return {
      visible: false,
      message: "",
      path: "ws://127.0.0.1:1594/shares/send_message_admin/",
      socket: "",
      token: localStorage.getItem('token'),
      history:[]
    };
  },
  methods: {
    showModal() {
      this.visible = true;
      this.init();
    },
    handleOk(e) {
      console.log(this.value);
      console.log(e);
      this.visible = false;
    },
    init: function () {
      if (typeof WebSocket === "undefined") {
        alert("您的浏览器不支持socket");
      } else {
        // 实例化socket
        this.socket = new WebSocket(this.path + '?token=' + this.token);
        // 监听socket连接
        this.socket.onopen = this.open;
        // 监听socket错误信息
        this.socket.onerror = this.error;
        // 监听socket消息
        this.socket.onmessage = this.getMessage;
      }
    },
    open: function (msg) {
      console.log("socket连接成功");
    },
    error: function () {
      console.log("连接错误");
    },
    getMessage: function (msg) {
      this.history = JSON.parse(msg.data)
      console.log("后端传来的数据", msg.data); // 打印后台返回的数据
    },
    sendMessage: function () {
      this.socket.send(this.message); // 发送给后台的数据
    },
    close: function () {
      console.log("socket已经关闭");
    },
  },
  destroyed() {
    // 销毁监听
    this.socket.onclose = this.close;
  },
  mounted() {},
};
</script>
